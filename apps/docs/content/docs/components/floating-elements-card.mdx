      ```tsx
      import { ClassValue, clsx } from 'clsx';
      import { twMerge } from 'tailwind-merge';

      export function cn(...inputs: ClassValue[]) {
          return twMerge(clsx(inputs));
      }
      ```

### Props

Use the following props to customize the floating elements and card content.

| Prop                | Type                     | Description                                        |
| :------------------ | :----------------------- | :------------------------------------------------- |
| <Callout>children</Callout>          | <Callout>React.ReactNode</Callout>          | The content to be displayed inside the main (front) card. |
| <Callout>icons</Callout>         | <Callout>`{ icon: React.ReactNode; link: string }[]`</Callout>         | Array of icon objects with icon element and link URL that float around the card on hover. |
