    </Steps>
  </TabsContent>

  <TabsContent value="manual">
    <Steps>
      <Step>Install dependencies</Step>

      ```bash
      npm install class-variance-authority motion clsx tailwind-merge
      ```
      <Step>Add util file</Step>
      <Callout>lib/utils.ts</Callout>

      ```tsx
      import { ClassValue, clsx } from 'clsx';
      import { twMerge } from 'tailwind-merge';

      export function cn(...inputs: ClassValue[]) {
          return twMerge(clsx(inputs));
      }
      ```

      <Step>Copy the source code</Step>
      <Callout>components/ui/ios-notifications-stack.tsx</Callout>

### Props

Use the following props to customize the floating elements and card content.

| Prop                | Type                     | Description                                        |
| :------------------ | :----------------------- | :------------------------------------------------- |
| <Callout>className</Callout>         | <Callout>string</Callout>                   | The CSS class to be applied to the card.           |
| <Callout>notifications</Callout>         | <Callout>`{ id: number, title: string; description: string }[]`</Callout>         | Array of notification objects containing a title and description to be displayed in the notification. |
